  8 /**
  9  * <p>
 10  *   This class describes the public APIs for the R Web Services. This module is
 11  *   intended as the primary client public interface to aid in the consumption of the R
 12  *   Web Services. All APIs invoke an asynchronous transaction where the response is
 13  *   recieved and managed within the respective callback method.
 14  * </p>
 15  * <p>
 16  * Typical useage: 
 17  * <pre>
 18  * var success = function(response) { // handle response success };
 19  * var fail = function(response) { // handle response error };
 20  * var callback = { success : success, failure: fail, scope : this, verbose : true };
 21  * // send request to create session, response in your callback
 22  * R.util.DeployR.createSession(callback);
 23  * </pre>
 24  *  </p>
 25  * <p><br/>Callback format: <a name="callback-format"></a></p>
 26  * <p> 
 27  * <ul>
 28  *   <li><code>success</code> - {Function} success callback</li>
 29  *   <li><code>failure</code>- {Function} failure callback</li>
 30  *   <li><code>scope</code> - {object} (Optional) callback execution scope.</li>
 31  *   <li><code>verbose</code> - {boolean} <code>true</code> to include the full
 32  *    successful sever respone, otherwise a subset of the response will be provided
 33  *    as defined by the this call. The default is <code>false</code>.
 34  *   </li>
 35  * </ul>    
 36  * </p>
 37  * The supported service categories within this module are defined as:
 38  * <p>
 39  * <ul>
 40  *  <li><code>Session Services</code> - 
 41  *    R sessions are created in response to explicit requests on the R Web services
 42  *    API such as Create Session or created implicitly to facilitate the stateless 
 43  *    execution of R scripts.
 44  *  </li>
 45  *  <li><code>Session Object Services</code> - 
 46  *    Defines the services for the mangement of Live objects in an R session.
 47  *  </li>
 48  *  <li><code>Session File Services</code> - 
 49  *    Defines services for the mangement of working directory files. 
 50  *  </li>
 51  *  <li><code>Repository Object Services</code> - 
 52  *    Defines services for the permanent store of a user's binary R objects across 
 53  *    diffrent R sessions.
 54  *  </li>
 55  *  <li><code>Repository File Services</code> -
 56  *    Defines the a permanent storeage repository services for user files. 
 57  *  </li>
 58  *  <li><code>Project Services</code> - 
 59  *    Defines the artifacts within a project that can be referenced within the context 
 60  *    of a live session for that project. 
 61  *  </li>
 62  *  <li><code>User Services</code> -
 63  *    Defines the diffrent user managment services involved in starting any R session
 64  *    within RevoDeployR.
 65  *  </li>
 66  *  <li><code>R Script Services</code> -
 67  *    Defines the service operations to access a collection of predefined R scripts. 
 68  *  </li>
 69  * </ul>
 70  * </p>
 71  *

157   /**
158    * A utility method to send off both successful and failure callbacks provided by 
159    * the caller. 
160    * @private
161    * @inner
162    * @param {boolean} isSucces Wether the API request was success (Status 200).
163    * @param {Object} response The response object as defined by the respect API
164    * category.
165    */
166   var fireCallbackResponse = function (isSuccess, response, callback) {
167 	if (callback) {
168       var scope = (callback.scope || this);
169       if (isSuccess && callback.success) {
170         callback.success.call(scope, response);
171       } else if (!isSuccess && callback.failure){
172         callback.failure.call(scope, response);
173       }
174       callback = null; 
175       scope = null;
176     }
177   };

184 			 }
185           }

196 			}
197          }    
198 	  } catch (e) {
199 		  value = def;
200 	  }
201 
202       return value;
203    };
